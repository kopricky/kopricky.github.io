Matching

コードについての説明

一般のグラフの最大マッチングを求める乱択アルゴリズム. 一般のグラフの最大マッチングは deterministic に多項式時間で解けることが知られている(Gallai, Edmonds) が, この実装では簡単な乱択アルゴリズムの実装をした.
辺 (u, v) が存在するなら T_uv = x_uv, T_vu = -x_uv(x_uv は変数), 存在しないなら T_uv = T_vu = 0 とするような行列を Tutte(タット) 行列と呼ぶ. 実はグラフが完全マッチングであることの必要十分条件はそのグラフに対応する Tutte 行列の行列式が恒等的に非零である.
しかし, Tutte 行列は変数を含んでいるためその行列式を効率よく計算することは困難である. そのため x_uv に異なる乱数を入れて何度か行列式の計算を繰り返し, 一度でも非零になったら完全マッチング, すべて零なら完全マッチングでないというような randomized を加えたものが本アルゴリズムである.
行列式が恒等式の意味では非零であるのに入れた乱数によって偶然 0 になってしまう確率は Shwarz-Zippel lemma を用いて評価することができ, その確率は乱数の台集合のサイズを |S|, 頂点数を n としたとき, d / |S| 以下であることがいえる.
完全マッチングを実際に求めたい場合は各辺についてその両端点の頂点を除いたときに完全マッチングであるかどうかの判定に影響が出るかどうかを見てやればよいので余分に O(m) factor の計算時間をかけると解ける.
また　Tutte(タット) 行列の rank はグラフの最大マッチングの 2 倍であることも知られているのでこちらも異なる乱数について何度か rank を計算し, 最頻値を取ることで正しい値を得ることができる.

(メンバ変数)
LOOP: ループの回数(100 回にしてるけど 10 回とかでも大丈夫なはず)
(関数)
add_edge(u, v): 辺(u, v) を追加する
perfect_matchching(): 完全マッチングであるかどうかを返す.
maximal_matchching(): 最大マッチングのサイズを返す.

時間計算量: O(n^3*LOOP) (LOOP は試行回数を表す)

コード

template<typename T> class mat : public vector<vector<T> > {
private:
    int r,c;    //行,列
public:
    inline int row() const {
        return r;
    }
    inline int column() const {
        return c;
    }
    mat(int n, int m, T val = 0){
        r = n, c = m;
        for(int i = 0; i < n; i++){
            this->push_back(vector<T>(m, val));
        }
    }
    int rank(){
        int res = 0;
        for(int i = 0; i < c; i++){
            if(res == r) return res;
            int pivot = res;
            for(int j = res + 1; j < r; j++){
                if(abs((*this)[j][i]) > abs((*this)[pivot][i])){
                    pivot = j;
                }
            }
            if(abs((*this)[pivot][i]) < EPS) continue;
            swap((*this)[pivot],(*this)[res]);
            for(int j = i + 1; j < c; j++){
                (*this)[res][j] /= (*this)[res][i];
            }
            for(int j = res + 1; j < r; j++){
                for(int k = i + 1; k < c; k++){
                    (*this)[j][k] -= (*this)[res][k]*(*this)[j][i];
                }
            }
            res++;
        }
        return res;
    }
    T det(){
        T ans = 1;
        mat B(r, r);
        for(int i = 0; i < r; i++){
            for(int j = 0; j < c; j++){
                B[i][j] = (*this)[i][j];
            }
        }
        for(int i = 0; i < c; i++){
            int pivot = i;
            for(int j = i + 1; j < r; j++){
                if(abs(B[j][i]) > abs(B[pivot][i])){
                    pivot = j;
                }
            }
            if(abs(B[pivot][i]) < EPS) return (T)0;
            if(pivot != i) swap(B[i], B[pivot]), ans = -ans;
            ans *= B[i][i];
            for(int j = i + 1; j < r; j++){
                for(int k = c - 1; k >= i; k--){
                    B[j][k] -= B[i][k] * B[j][i] / B[i][i];
                }
            }
        }
        return ans;
    }
    inline void print() const {
        for(int i = 0; i < r; i++){
            for(int j = 0; j < (c-1); j++){
                cout << (*this)[i][j] << ",";
            }
            cout << (*this)[i][c-1] << endl;
        }
    }
};
 
class Matching {
private:
    int V;
    vector<pair<int, int> > es;
    vector<int> cnt;
    static constexpr int LOOP = 100;
    unsigned long rnd(){
        static unsigned long long x = 123456789, y = 362436069, z = 521288629, w = 86675123;
        unsigned long long t = (x^(x<<11));
        x = y,y = z,z = w;
        return (w = (w^(w>>19))^(t^(t>>8)));
    }
     
public:
    Matching(int node_size) : V(node_size), cnt(V+1, 0){}
    void add_edge(int u, int v){
        if(u > v) swap(u, v);
        es.push_back(make_pair(u, v));
    }
    bool perfect_matchching(){
        for(int i = 0; i < LOOP; i++){
            mat<double> A(V, V);
            for(auto e : es){
                double r = (double)rnd()/((double)ULONG_MAX+1);
                A[e.first][e.second] = r, A[e.second][e.first] = -r;
            }
            if(abs(A.det()) > EPS) return true;
        }
        return false;
    }
    int maximal_matchching(){
        for(int i = 0; i < LOOP; i++){
            mat<double> A(V, V);
            for(auto e : es){
                double r = (double)rnd()/((double)ULONG_MAX+1);
                A[e.first][e.second] = r, A[e.second][e.first] = -r;
            }
            cnt[A.rank()]++;
        }
        return (int)(max_element(cnt.begin(), cnt.end()) - cnt.begin()) / 2;
    }
};
