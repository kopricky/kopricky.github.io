Update Interval

コードについての説明

Segment Tree で持てないような大きな範囲についての範囲の 0,1(ON/OFF) を管理する. 1 (ON) の範囲を set<pair<int, int> > で持つ.
また範囲 [1,3), [3,5) などはちゃんと [1,5) にマージされるように実装している.
たまに使えるシーンがある.

時間計算量: 更新 ならしO(logn)

コード

template<typename T> class UpdateInterval
{
public:
using ptt = pair<T,T>;
    // 1の区間をsetで保持
    set<ptt> st;
    UpdateInterval(){};
    // [l,r) を k ({0,1})に更新
    void update(T l,T r,int k){
        T lb = l, rb = r;
        while(1){
            auto it = st.lower_bound(ptt(l, numeric_limits<T>::min()));
            if(it == st.end()){
                break;
            }
            ptt p = *it;
            if(p.fi > r){
                break;
            }
            st.erase(it);
            if(p.se > r){
                rb = max(rb, p.se);
                if(!k) st.insert(ptt(r,p.se));
                break;
            }
        }
        auto it = st.lower_bound(ptt(l, numeric_limits<T>::min()));
        if(it != st.begin()){
            --it;
            ptt p = *it;
            if(p.se >= l){
                st.erase(it);
                if(!k) st.insert(ptt(p.fi,l));
                lb = min(lb, p.fi);
            }
            if(p.se > r){
                if(!k) st.insert(ptt(r,p.se));
                rb = max(rb, p.se);
            }
        }
        if(k) st.insert(ptt(lb,rb));
    }
};
