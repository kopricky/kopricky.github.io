Hungarian

コードについての説明

割当問題を効率よく解くアルゴリズム. 割当問題とは n 人の人と n 個の仕事があり, また各(人 i, 仕事 j) のペアについて人 i が仕事 j を行うのにかかる時間が与えられているとする. このとき n 人の人をちょうど 1 つずつ別々の仕事に割り当てることで仕事にかかるのべ時間を最小にしたいという問題のことを言う.
二部グラフのマッチングの重みつきの場合のアルゴリズムである. 最小費用流のアルゴリズムでも解くことができるが, こちらだと計算量としては O(n^3 log n) かかるのに対し, ハンガリアン法では O(n^3) の計算量で解くことが可能である. また学問的に割合問題は重み付き(線形)マトロイド交差問題の枠組みで解かれることが多く, 重みが小さい場合には効率よく動作するアルゴリズムも知られている.
Hungarian 法の説明は有名な Goemans 先生の Lecture note が分かりやすかった. 以上の議論を辺コストを要素とする行列上でみるとよく見るハンガリアン法の説明の行、 列最小の値引いて割り当てを見つけて更新して...みたいなことをすればできるという話になる. これだと正直何をやってるいるかが分かりにくいので速度は劣ると思うが増加道アルゴリズムによる実装を行った.
ちなみに Hungarian はこの問題に対する重要な貢献をした 2 人のハンガリー人数学者 Kőnig, Egerváry にちなんで呼ばれている.

(関数)
solve(cost): n×n の辺コストを要素とする配列を渡し, (総コスト, 割当) を返す.

時間計算量: O(n^3)

コード

// グラフが完全二部グラフであることを仮定しています.
template<typename T>
class Hungarian
{
private:
    int V, matching_size;
    vector<vector<bool> > graph;
    vector<vector<int> > residue;
    vector<int> potential, allocate;
    vector<bool> right_alloc;
    T total_cost;
    void initial_feasible_solution(const vector<vector<T> >& cost){
        for(int i = 0; i < V; i++){
            potential[i] = 0;
            for(int j = V; j < 2*V; j++){
                potential[j] = min(potential[j], cost[i][j-V]);
            }
        }
    }
    void inital_graph_construction(const vector<vector<T> >& cost){
        for(int i = 0; i < V; i++){
            for(int j = V; j < 2*V; j++){
                residue[i][j-V] = cost[i][j-V] - potential[i] - potential[j];
                graph[i][j-V] = (residue[i][j-V] == 0);
            }
        }
    }
    bool update(){
        vector<int> visit(2*V, -1);
        if(increment_matching(visit)) return true;
        for(int i = 0; i < V; i++){
            if(allocate[i] < 0 && visit[i] < 0) dfs(i, i, visit);
        }
        T delta = std::numeric_limits<T>::max();
        for(int i = 0; i < V; i++){
            if(visit[i] < 0) continue;
            for(int j = V; j < 2*V; j++){
                if(visit[j] < 0){
                    delta = min(delta, residue[i][j-V]);
                }
            }
        }
        update_potential(visit, delta);
        update_graph();
        return false;
    }
    void dfs(const int cur, const int start, vector<int>& visit){
        visit[cur] = start;
        for(int i = 0; i < V; i++){
            if(graph[cur][i] && visit[i + (cur < V)*V] < 0){
                dfs(i + (cur < V)*V, start, visit);
            }
        }
    }
    void update_potential(const vector<int>& visit, const T delta){
        for(int i = 0; i < V; i++){
            if(visit[i] >= 0){
                potential[i] += delta;
                for(int j = 0; j < V; j++){
                    residue[i][j] -= delta;
                }
            }
        }
        for(int j = V; j < 2*V; j++){
            if(visit[j] >= 0){
                potential[j] -= delta;
                for(int i = 0; i < V; i++){
                    residue[i][j-V] += delta;
                }
            }
        }
    }
    void update_graph(){
        for(int i = 0; i < V; i++){
            for(int j = 0; j < V; j++){
                graph[i][j] = ((residue[i][j] == 0) && !graph[j+V][i]);
            }
        }
    }
    bool increment_matching(vector<int>& visit){
        for(int i = 0; i < V; i++){
            if(allocate[i] < 0){
                vector<int> path;
                if(find_augmenting_path(i, path, visit)){
                    matching_size++;
                    for(int j = 0; j < (int)path.size() - 1; j++){
                        if(j % 2){
                            graph[path[j]][path[j+1]] = false;
                            graph[path[j+1]][path[j]-V] = true;
                        }else{
                            graph[path[j]][path[j+1]-V] = false;
                            graph[path[j+1]][path[j]] = true;
                            allocate[path[j]] = path[j+1]-V;
                            right_alloc[path[j+1]-V] = true;
                        }
                    }
                    if(matching_size == V){
                        return true;
                    }
                }
                for(int j = 0; j < 2*V; j++) visit[j] = -1;
            }
        }
        return false;
    }
    bool find_augmenting_path(const int cur, vector<int>& path, vector<int>& visit){
        path.push_back(cur);
        visit[cur] = 0;
        if(cur >= V && !right_alloc[cur-V]){
            return true;
        }
        for(int i = 0; i < V; i++){
            if(graph[cur][i] && visit[i + (cur < V)*V] < 0){
                if(find_augmenting_path(i + (cur < V)*V, path, visit)){
                    return true;
                }
            }
        }
        path.pop_back();
        return false;
    }
public:
    Hungarian() : matching_size(0), total_cost(0){}
    pair<T, vector<int> > solve(const vector<vector<T> >& cost){
        V = (int)cost.size();
        graph.resize(2*V, vector<bool>(V, false));
        residue.resize(V, vector<int>(V));
        potential.resize(2*V, std::numeric_limits<T>::max());
        allocate.resize(V, -1);
        right_alloc.resize(V, false);
        initial_feasible_solution(cost);
        inital_graph_construction(cost);
        while(true){
            if(update()) break;
        }
        for(int i = 0; i < V; i++){
            total_cost += cost[i][allocate[i]];
        }
        return make_pair(total_cost, allocate);
    }
};
