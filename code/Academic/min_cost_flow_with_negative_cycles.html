Minimum Cost Flow Algorithm with Negative Cycles

コードについての説明(個人的メモ)

負の有向閉路を含むグラフに対する最小費用流問題を解くアルゴリズム. 正確には任意のコスト関数, 容量関数に対して正しい $s$-$t$ 間最小費用流を返すアルゴリズムである. 複数の source および sink がある場合も超頂点 $S$, $T$ を用意して $S$ から各 source, 各 sink から $T$ へ辺を張ることで変換可能.
このアルゴリズムは弱多項式時間であるだけでなく, 強多項式時間であることも後の解析で明らかになり, 計算量は $\O (\min (m^2 n^2 \log (n |c_{\min}|), m^3 n^2 \log n))$ となる($c_{\min}$ は辺コストの最小値を表し, また前者は辺のコストが整数の場合にのみ成り立つ).
実測は速くなく他の方の実装のほうが速いと思う. ただ通常上記の計算量の見積りよりかなり速く動作する.

時間計算量: $\O (\min (m^2 n^2 \log (n |c_{\min}|), m^3 n^2 \log n))$ (ただし $c_{\min}$ は辺コストの最小値を表し, また前者は辺のコストが整数の場合にのみ成り立つ)

コード

template<class edge> class Dinic {
private:
    using Cat = decltype(edge::cap);
    using Cot = decltype(edge::cost);
    vector<vector<edge> >& G;
    const int V;
    vector<int> level, iter;
    void bfs(const int s){
        fill(level.begin(), level.end(), -1);
        queue<int> que;
        level[s] = 0;
        que.push(s);
        while(!que.empty()){
            const int v = que.front();
            que.pop();
            for(const auto& e : G[v]){
                if(e.cap > 0 && level[e.to] < 0){
                    level[e.to] = level[v] + 1;
                    que.push(e.to);
                }
            }
        }
    }
    Cat dfs(const int v, const int t, const Cat f, Cot& res){
        if(v == t) return f;
        for(int& i = iter[v]; i < (int)G[v].size(); ++i){
            edge& e = G[v][i];
            if(e.cap > 0 && level[v] < level[e.to]){
                Cat d = dfs(e.to, t, min(f, e.cap), res);
                if(d > 0){
                    e.cap -= d;
                    G[e.to][e.rev].cap += d;
                    res += d * e.cost;
                    return d;
                }
            }
        }
        return 0;
    }

public:
    Dinic(vector<vector<edge> >& _G) : G(_G), V((int)G.size()), level(V), iter(V){}
    Cot solve(const int s, const int t, Cat f){
        Cot res = 0;
        for(;;){
            bfs(s);
            if(level[t] < 0) return numeric_limits<Cot>::min();
            fill(iter.begin(), iter.end(), 0);
            while(true){
                Cat df = dfs(s, t, f, res);
                if(df > 0){
                    f -= df;
                    if(f == 0) return res;
                }else{
                    break;
                }
            }
        }
    }
};

template<typename CapType, typename CostType> class MinCostFlow
{
public:
    struct edge;
private:
    using Cat = CapType;
    using Cot = CostType;
    const int V;
    const Cot inf;
    vector<vector<Cot> > dp;
    vector<vector<edge*> > prev;
    int minimum_mean_cycle(){
        for(int k = 0; k < V; ++k){
            for(int i = 0; i < V; ++i){
                dp[k+1][i] = inf;
            }
            for(int i = 0; i < V; ++i){
                if(dp[k][i] == inf) continue;
                for(edge& e : G[i]){
                    if(e.cap > 0 && dp[k+1][e.to] > dp[k][i] + e.cost){
                        dp[k+1][e.to] = dp[k][i] + e.cost;
                        prev[k+1][e.to] = &e;
                    }
                }
            }
        }
        if(*min_element(dp[V].begin(), dp[V].end()) == inf) return -1;
        int start = -1;
        Cot tnum = inf / V;
        int tden = 1;
        for(int i = 0; i < V; ++i){
            if(dp[V][i] == inf) continue;
            Cot num = -inf / V;
            int den = 1;
            for(int k = 0; k < V; ++k){
                if(dp[k][i] == inf) continue;
                if(num * (V-k) <= den * (dp[V][i] - dp[k][i])){
                    num = dp[V][i] - dp[k][i], den = V - k;
                }
            }
            if(num * tden < den * tnum){
                tnum = num, tden = den, start = i;
            }
        }
        return (tnum < 0) ? start : -1;
    }
    bool update(Cot& res){
        int start = minimum_mean_cycle();
        if(start < 0) return false;
        int level = V;
        Cat df = numeric_limits<Cat>::max();
        vector<Cat> route;
        vector<int> used(V, -1);
        while(true){
            if(used[start] >= 0){
                for(int i = used[start]; i < (int)route.size(); ++i){
                    df = min(df, route[i]);
                }
                level = V - used[start];
                break;
            }
            used[start] = (int)route.size();
            edge *e = prev[level--][start];
            route.push_back(e->cap);
            start = G[e->to][e->rev].to;
        }
        int cur = start;
        do {
            edge *e = prev[level--][cur];
            edge *reve = &G[e->to][e->rev];
            e->cap -= df, reve->cap += df, res += df * e->cost;
            cur = reve->to;
        }while(level >= 0 && cur != start);
        return true;
    }
public:
    struct edge {
        int to; Cat cap; Cot cost; int rev;
    };
    vector<vector<edge> > G;
    MinCostFlow(const int node_size)
     : V(node_size), inf(numeric_limits<Cot>::max()),
        dp(V+1, vector<Cat>(V, 0)), prev(V+1, vector<edge*>(V)), G(V){}
    void add_edge(const int from, const int to, const Cat cap, const Cot cost){
        G[from].push_back((edge){to, cap, cost, (int)G[to].size()});
        G[to].push_back((edge){from, 0, -cost, (int)G[from].size() - 1});
    }
    Cot solve(const int s, const int t, Cat f){
        Dinic<edge> dn(G);
        Cot res = dn.solve(s, t, f);
        if(res == numeric_limits<Cot>::min()) return res;
        while(update(res));
        return res;
    }
};
