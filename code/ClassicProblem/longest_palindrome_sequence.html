int LPS(const string& s)
{
    int n = (int)s.size();
    vector<vector<int> > dp(n,vector<int>(n+1,0));
    rep(i,n){
        dp[i][i+1] = 1;
    }
    //iは部分文字列の長さ
    for(int i = 2; i <= n; i++){
        rep(j,n-i+1){
            int k = i + j;
            if(s[j] == s[k-1]){
                dp[j][k] = dp[j+1][k-1] + 2;
            }else{
                dp[j][k] = max(dp[j][k-1], dp[j+1][k]);
            }
        }
    }
    return dp[0][n];
